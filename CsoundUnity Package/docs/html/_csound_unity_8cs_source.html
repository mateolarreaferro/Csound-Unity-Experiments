503  public static List<string> ParseCsdFileForAudioChannels(string filename)
504  {
505  if (!File.Exists(filename)) return null;
506 
507  string[] fullCsdText = File.ReadAllLines(filename);
508  if (fullCsdText.Length < 1) return null;
509 
510  List<string> locaAudioChannels = new List<string>();
511 
512  foreach (string line in fullCsdText)
513  {
514  var trimmd = line.TrimStart();
515  if (!trimmd.Contains("chnset")) continue;
516  if (trimmd.StartsWith(";")) continue;
517  var lndx = trimmd.IndexOf("chnset");
518  var chnsetEnd = lndx + "chnset".Length + 1;
519  var prms = trimmd.Substring(chnsetEnd, trimmd.Length - chnsetEnd);
520  var split = prms.Split(',');
521  if (!split[0].StartsWith("a") && !split[0].StartsWith("ga"))
522  continue; //discard non audio variables
523  // Debug.Log("found audio channel");
524  var ach = split[1].Replace('\\', ' ').Replace('\"', ' ').Trim();
525  if (!locaAudioChannels.Contains(ach))
526  locaAudioChannels.Add(ach);
527  }
528  return locaAudioChannels;
529  }

536  public static List<CsoundChannelController> ParseCsdFile(string filename)
537  {
538  if (!File.Exists(filename)) return null;
539 
540  string[] fullCsdText = File.ReadAllLines(filename);
541  if (fullCsdText.Length < 1) return null;
542 
543  List<CsoundChannelController> locaChannelControllers;
544  locaChannelControllers = new List<CsoundChannelController>();
545 
546  foreach (string line in fullCsdText)
547  {
548 
549  if (line.Contains("</"))
550  break;
551 
552  var trimmd = line.TrimStart();
553  //discard csound comments in cabbage widgets
554  if (trimmd.StartsWith(";"))
555  {
556  //Debug.Log("discarding "+line);
557  continue;
558  }
559  string newLine = trimmd;
560  string control = trimmd.Substring(0, trimmd.IndexOf(" ") > -1 ? trimmd.IndexOf(" ") : 0);
561  if (control.Length > 0)
562  newLine = newLine.Replace(control, "");
563 
564  if (control.Contains("slider") || control.Contains("button") || control.Contains("checkbox")
565  || control.Contains("groupbox") || control.Contains("form") || control.Contains("combobox"))
566  {
567  CsoundChannelController controller = new CsoundChannelController();
568  controller.type = control;
569 
570  if (trimmd.IndexOf("caption(") > -1)
571  {
572  string infoText = trimmd.Substring(trimmd.IndexOf("caption(") + 9);
573  infoText = infoText.Substring(0, infoText.IndexOf(")") - 1);
574  controller.caption = infoText;
575  }
576 
577  if (trimmd.IndexOf("text(") > -1)
578  {
579  string text = trimmd.Substring(trimmd.IndexOf("text(") + 6);
580  text = text.Substring(0, text.IndexOf(")") - 1);
581  text = text.Replace("\"", "");
582  text = text.Replace('"', new char());
583  if (controller.type == "combobox") //if combobox, create a range
584  {
585  char[] delimiterChars = { ',' };
586  string[] tokens = text.Split(delimiterChars);
587  controller.SetRange(1, tokens.Length, 0);
588 
589  for (var o = 0; o < tokens.Length; o++)
590  {
591  tokens[o] = string.Join("", tokens[o].Split(default(string[]), System.StringSplitOptions.RemoveEmptyEntries));
592  }
593  controller.options = tokens;
594  }
595  }
596 
597  if (trimmd.IndexOf("items(") > -1)
598  {
599  string text = trimmd.Substring(trimmd.IndexOf("items(") + 7);
600  text = text.Substring(0, text.IndexOf(")") - 1);
601  //TODO THIS OVERRIDES TEXT!
602  text = text.Replace("\"", "");
603  text = text.Replace('"', new char());
604  if (controller.type == "combobox")
605  {
606  char[] delimiterChars = { ',' };
607  string[] tokens = text.Split(delimiterChars);
608  controller.SetRange(1, tokens.Length, 0);
609 
610  for (var o = 0; o < tokens.Length; o++)
611  {
612  tokens[o] = string.Join("", tokens[o].Split(default(string[]), System.StringSplitOptions.RemoveEmptyEntries));
613  }
614  controller.options = tokens;
615  }
616  }
617 
618  if (trimmd.IndexOf("channel(") > -1)
619  {
620  string channel = trimmd.Substring(trimmd.IndexOf("channel(") + 9);
621  channel = channel.Substring(0, channel.IndexOf(")") - 1);
622  controller.channel = channel;
623  }
624 
625  if (trimmd.IndexOf("range(") > -1)
626  {
627  int rangeAt = trimmd.IndexOf("range(");
628  if (rangeAt != -1)
629  {
630  string range = trimmd.Substring(rangeAt + 6);
631  range = range.Substring(0, range.IndexOf(")"));
632  char[] delimiterChars = { ',' };
633  string[] tokens = range.Split(delimiterChars);
634  for (var i = 0; i < tokens.Length; i++)
635  {
636  tokens[i] = string.Join("", tokens[i].Split(default(string[]), StringSplitOptions.RemoveEmptyEntries));
637  if (tokens[i].StartsWith("."))
638  {
639  tokens[i] = "0" + tokens[i];
640  }
641  if (tokens[i].StartsWith("-."))
642  {
643  tokens[i] = "-0" + tokens[i].Substring(2, tokens[i].Length - 2);
644  }
645  }
646  var min = float.Parse(tokens[0], CultureInfo.InvariantCulture);
647  var max = float.Parse(tokens[1], CultureInfo.InvariantCulture);
648  var val = float.Parse(tokens[2], CultureInfo.InvariantCulture);
649  controller.SetRange(min, max, val);
650  }
651  }
652 
653  if (line.IndexOf("value(") > -1)
654  {
655  string value = trimmd.Substring(trimmd.IndexOf("value(") + 6);
656  value = value.Substring(0, value.IndexOf(")"));
657  value = value.Replace("\"", "");
658  controller.value = value.Length > 0 ? float.Parse(value, CultureInfo.InvariantCulture) : 0;
659  if (control.Contains("combobox"))
660  {
661  //Cabbage combobox index starts from 1
662  controller.value = controller.value - 1;
663  // Debug.Log("combobox value in parse: " + controller.value);
664  }
665  }
666  locaChannelControllers.Add(controller);
667  }
668  }
669  return locaChannelControllers;
670  }

1414  private void ProcessBlock(float[] samples, int numChannels)
1415  {
1416  if (compiledOk && initialized && !_quitting)
1417  {
1418  for (int i = 0; i < samples.Length; i += numChannels, ksmpsIndex++)
1419  {
1420  for (uint channel = 0; channel < numChannels; channel++)
1421  {
1422  if (mute == true)
1423  samples[i + channel] = 0.0f;
1424  else
1425  {
1426  if ((ksmpsIndex >= GetKsmps()) && (GetKsmps() > 0))
1427  {
1428  var res = PerformKsmps();
1429  performanceFinished = res == 1;
1430  ksmpsIndex = 0;
1431 
1432  foreach (var chanName in availableAudioChannels)
1433  {
1434  if (!namedAudioChannelTempBufferDict.ContainsKey(chanName)) continue;
1435  namedAudioChannelTempBufferDict[chanName] = GetAudioChannel(chanName);
1436  }
1437  }
1438 
1439  if (processClipAudio)
1440  {
1441  SetInputSample((int)ksmpsIndex, (int)channel, samples[i + channel] * zerdbfs);
1442  }
1443 
1444  //if csound nChnls are more than the current channel, set the last csound channel available on the sample (assumes GetNchnls above 0)
1445  var outputSampleChannel = channel < GetNchnls() ? channel : GetNchnls() - 1;
1446  //var rand = new System.Random();
1447  //samples[i + channel] = (float)rand.NextDouble();
1448  samples[i + channel] = (float)GetOutputSample((int)ksmpsIndex, (int)outputSampleChannel) / zerdbfs;
1449 
1450  if (loudVolumeWarning && (samples[i + channel] > loudWarningThreshold))
1451  {
1452  samples[i + channel] = 0.0f;
1453  Debug.LogWarning("Volume is too high! Clearing output");
1454  }
1455  }
1456  }
1457 
1458  // update the audioChannels just when this instance is not muted
1459  if (!mute)
1460  foreach (var chanName in availableAudioChannels)
1461  {
1462  if (!namedAudioChannelDataDict.ContainsKey(chanName) || !namedAudioChannelTempBufferDict.ContainsKey(chanName)) continue;
1463  namedAudioChannelDataDict[chanName][i / numChannels] = namedAudioChannelTempBufferDict[chanName][ksmpsIndex];
1464  }
1465  }
1466  }
1467  }
