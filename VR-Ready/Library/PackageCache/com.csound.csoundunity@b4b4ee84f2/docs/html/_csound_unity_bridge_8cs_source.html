65  //Csound6.NativeMethods.csoundSetGlobalEnv("SADIR", Application.persistentDataPath);

621  public IDictionary<string, IList<OpcodeArgumentTypes>> GetOpcodeList()
622  {
623  var opcodes = new SortedDictionary<string, IList<OpcodeArgumentTypes>>();
624  IntPtr ppOpcodeList = IntPtr.Zero;
625  int size = Csound6.NativeMethods.csoundNewOpcodeList(csound, out ppOpcodeList);
626  if ((ppOpcodeList != IntPtr.Zero) && (size >= 0))
627  {
628  int proxySize = Marshal.SizeOf(typeof(OpcodeListProxy));
629  for (int i = 0; i < size; i++)
630  {
631  OpcodeListProxy proxy = Marshal.PtrToStructure(ppOpcodeList + (i * proxySize), typeof(OpcodeListProxy)) as OpcodeListProxy;
632  string opname = Marshal.PtrToStringAnsi(proxy.opname);
633  OpcodeArgumentTypes opcode = new OpcodeArgumentTypes
634  {
635  outypes = Marshal.PtrToStringAnsi(proxy.outtypes),
636  intypes = Marshal.PtrToStringAnsi(proxy.intypes),
637  flags = proxy.flags
638  };
639  if (!opcodes.ContainsKey(opname))
640  {
641  IList<OpcodeArgumentTypes> types = new List<OpcodeArgumentTypes>();
642  types.Add(opcode);
643  opcodes.Add(opname, types);
644  }
645  else
646  {
647  opcodes[opname].Add(opcode);
648  }
649  }
650  Csound6.NativeMethods.csoundDisposeOpcodeList(csound, ppOpcodeList);
651  }
652  return opcodes;
653  }

682  public IDictionary<string, ChannelInfo> GetChannelList()
683  {
684  IDictionary<string, ChannelInfo> channels = new SortedDictionary<string, ChannelInfo>();
685  IntPtr ppChannels = IntPtr.Zero;
686  int size = Csound6.NativeMethods.csoundListChannels(csound, out ppChannels);
687  if ((ppChannels != IntPtr.Zero) && (size >= 0))
688  {
689  int proxySize = Marshal.SizeOf(typeof(ChannelInfoProxy));
690  for (int i = 0; i < size; i++)
691  {
692  var proxy = Marshal.PtrToStructure(ppChannels + (i * proxySize), typeof(ChannelInfoProxy)) as ChannelInfoProxy;
693  string chanName = Marshal.PtrToStringAnsi(proxy.name);
694  ChannelInfo info = new ChannelInfo(chanName, (ChannelType)(proxy.type & 15), (ChannelDirection)(proxy.type >> 4));
695  var hintProxy = proxy.hints;
696  var hints = new ChannelHints((ChannelBehavior)hintProxy.behav, hintProxy.dflt, hintProxy.min, hintProxy.max)
697  {
698  x = hintProxy.x,
699  y = hintProxy.y,
700  height = hintProxy.height,
701  width = hintProxy.width,
702  attributes = (hintProxy.attributes != IntPtr.Zero) ? Marshal.PtrToStringAnsi(hintProxy.attributes) : null
703  };
704  info.Hints = hints;
705  channels.Add(chanName, info);
706  }
707  Csound6.NativeMethods.csoundDeleteChannelList(csound, ppChannels);
708  }
709
